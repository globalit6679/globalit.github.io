void setup(){
    Serial.begin(115200);
    pinMode(A0, INPUT_PULLUP);
}

void loop(){
    int iLight = analogRead(A0);
    Serial.print("[");
    Serial.print(millis());
    Serial.print(",");
    Serial.print(iLight);
    Serial.println("]");
    delay(1000);
}
